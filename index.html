Usage

          
# Validate a rules file
python3 validate_snort3.py converted_rules.rules

# Read from stdin
cat converted_rules.rules | python3 validate_snort3.py

# Strict mode — treat warnings as errors (non-zero exit)
python3 validate_snort3.py --strict converted_rules.rules
